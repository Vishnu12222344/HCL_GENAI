<script>

    const baseurl = "https://www.themealdb.com/api/json/v1/1/filter.php?f=a";

    async function getMealByFirstLetter(x) {
        try {
            const response = await fetch(${baseurl}${x});
            if (!response.ok) throw new Error('Network response was not ok');
            const data = await response.json();
            const meals = data.meals;
            let output = <h2>Meals starting with '${x}':</h2><ul>;
            if (Array.isArray(meals) && meals.length > 0) {
                for (const meal of meals) {
                    output += <li>${meal.strMeal}</li>;
                }
            } else {
                output += '<li>No meals found.</li>';
            }
            output += '</ul>';
            document.getElementById('mealResults').innerHTML = output;
        } catch (error) {
            document.getElementById('mealResults').innerHTML = <p>Error: ${error.message}</p>;
        }
    }

    getMealByFirstLetter('a');
</script>
</html>
